<input id="precio" placeholder="Precio"><br><br>


<button onclick="guardar()">Guardar Factura</button>


<script>


function guardar() {


  const file = document.getElementById("foto").files[0];


  if(!file){

    alert("Toma una foto primero");

    return;

  }


  const reader = new FileReader();


  reader.onloadend = function () {


    fetch("https://script.google.com/macros/s/AKfycbzkgn0OCA_P9M3Ns8pCcpd6gpbu5iUPwH863ekl5hwgvJDVTz623GI6L1LMENbv0psf/exec", {

      method: "POST",

      body: JSON.stringify({

        item: "Factura",

        descripcion: "Compra",

        cantidad: 1,

        precio: document.getElementById("precio").value,

        comercio: document.getElementById("comercio").value,

        foto: reader.result

      })

    })

    .then(() => alert("Factura guardada ✅"));

  };


  reader.readAsDataURL(file);

}


</script>
